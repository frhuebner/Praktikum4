\newcommand{\hantel}[3]{%
  \draw[rotate around={#3:(#1,#2)}] (#1,#2) circle (0.2cm);
  \draw[rotate around={#3:(#1,#2)}] (#1+0.2,#2)--(#1+1.9,#2);
}

\newcommand{\lowerhantel}[3]{%
  \draw[rotate around={#3:(#1,#2)},fill=black] (#1,#2) circle (0.1cm);
  \draw[rotate around={#3:(#1,#2)}] (#1+0.1,#2)--(#1+2,#2);
}
