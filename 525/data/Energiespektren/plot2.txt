set terminal postscript eps enhanced size 3in,3in
set samples 1000

set xlabel 'E/keV'
set ylabel 'n'

set output 'energiekal_links.eps'

h(x)=a*x+b
fit h(x) 'energiekal_links.txt' u 1:2:3 yerr via a,b
p 'energiekal_links.txt' u 1:2:3 w yerrorbars title 'Messwerte', h(x) title 'Regr.'

set output 'energiekal_rechts.eps'

h(x)=a*x+b
fit h(x) 'energiekal_rechts.txt' u 1:2:3 yerr via a,b
p 'energiekal_rechts.txt' u 1:2:3 w yerrorbars title 'Messwerte', h(x) title 'Regr.'
